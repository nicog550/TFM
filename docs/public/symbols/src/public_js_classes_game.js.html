1 "use strict";
  2 /**
  3  * Class responsible for the game logic
  4  * @constructor
  5  */
  6 var Game = function() {
  7     var debugGame = false,
  8         gameOver,
  9         main,
 10         otherPlayersBoard = new OtherPlayersBoard(debugGame),
 11         playerBoard = new PlayerBoard(debugGame, setPlayerConfig, otherPlayersBoard.getBoard()),
 12         playerConfig,
 13         sockets,
 14         waitingRoom;
 15     return {
 16         init: function(mainRef, socketsRef, gameOverRef, waitingRoomRef) {
 17             main = mainRef;
 18             sockets = socketsRef;
 19             gameOver = gameOverRef;
 20             waitingRoom = waitingRoomRef;
 21             playerBoard.listenToMoves(sockets);
 22             gameOver.init(main, socketsRef);
 23         },
 24         debug: function() { return debugGame; },
 25         selector: "game-screen",
 26         drawMove: otherPlayersBoard.drawMove,
 27         finishGame: finishGame,
 28         startGame: startGame
 29     };
 30 
 31     /**
 32      * Prepares the game over screen and switches to it
 33      * @param {number} waitingTime The amount of time until the next game starts
 34      */
 35     function finishGame(waitingTime) {
 36         if (debugGame) return [];
 37         gameOver.setup(waitingTime);
 38         main.toggleScreen(gameOver);
 39     }
 40 
 41     /** Setter */
 42     function setPlayerConfig(index, value) {
 43         playerConfig[index] = parseInt(value);
 44     }
 45 
 46     /**
 47      * Displays the current player's board as well as the ones of the other players
 48      * @param {array} board The initial configuration for the current player
 49      * @param {number} duration The game duration
 50      * @param {number} options The number of different choices available at the board for the player
 51      * @param {string} myName The current player's name
 52      * @param {object} otherPlayers The initial configuration for each of the other players
 53      * @param {number} showCodeTime The amount of time the code will be shown to the player
 54      */
 55     function startGame(board, duration, options, myName, otherPlayers, showCodeTime) {
 56         playerConfig = board;
 57         playerBoard.drawBoard(board, options, myName);
 58         otherPlayersBoard.displayBoards(otherPlayers);
 59         var $remainingTime = $(".remaining-time");
 60         $remainingTime.css('visibility', 'hidden');
 61         main.toggleScreen(this);
 62         main.displayCountdown(showCodeTime, $(".initial-countdown"));
 63         //When the initial countdown ends, allow the player to start moving
 64         setTimeout(function() {
 65             main.displayCountdown(duration, $("#game-time"));
 66             $remainingTime.css('visibility', 'visible');
 67             playerBoard.allowMoves();
 68         }, showCodeTime * 1000);
 69     }
 70 };
