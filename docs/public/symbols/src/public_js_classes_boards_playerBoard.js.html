1 "use strict";
  2 /**
  3  * Class responsible for drawing the current player's board
  4  * @param {boolean} debugGame Game mode: debug or not
  5  * @param {function} setPlayerConfig Reference to the setter of the playerConfig variable
  6  * @param {jQuery} $otherPlayersBoard The container of the other players boards
  7  * @constructor
  8  */
  9 var PlayerBoard = function(debugGame, setPlayerConfig, $otherPlayersBoard) {
 10     var $gameBoardContainer = $("#own-game-container"),
 11         detachedChoices;
 12     return {
 13         allowMoves: allowMoves,
 14         drawBoard: drawBoard,
 15         listenToMoves: function(sockets) {
 16             performMove();
 17             submitMove(sockets);
 18         }
 19     };
 20 
 21     /**
 22      * When a new game starts, the player is shown the board for a period of time and cannot perform any moves. This
 23      * function allows the player to start moving
 24      */
 25     function allowMoves() {
 26         detachedChoices.forEach(function(current) {
 27             current.button.attr('data-background', '-1');
 28             current.button.removeClass('disabled');
 29             current.button.parent().append(current.choices);
 30         });
 31         $gameBoardContainer.find(".initial-countdown").remove();
 32         $gameBoardContainer.find(".submit-move").removeClass('hidden');
 33     }
 34 
 35     /**
 36      * Empties the board if it already had content and populates it again
 37      * @param {Array} board The values for the player's board (initial configuration)
 38      * @param {number} options The number of different choices available at the board for the player
 39      * @param {string} myName The player's name
 40      */
 41     function drawBoard(board, options, myName) {
 42         if (debugGame && $gameBoardContainer.children().length > 0) return;
 43         detachedChoices = [];
 44         $gameBoardContainer.empty();
 45         var $boardTemplate = $("#board-template").find(".own-game").clone(),
 46             $rowTemplate = $boardTemplate.find(".game-board"),
 47             $boxTemplate = $rowTemplate.find(".box").detach();
 48         $boardTemplate.find(".name").text(myName);
 49         for (var i = 0; i < board.length; i++) {
 50             $rowTemplate.append(_createDropdown($boxTemplate, board[i], i, options));
 51         }
 52         $gameBoardContainer.append($boardTemplate);
 53     }
 54 
 55     /**
 56      * Creates the list of choices for the player
 57      * @param {jQuery} $baseTemplate The template for creating a box with value plus the available choices
 58      * @param {number} boardValue The value for the current box
 59      * @param {number} boardPosition The position of the current box at the board
 60      * @param {number} options The number of different choices available at the board for the player
 61      * @returns {jQuery}
 62      * @private
 63      */
 64     function _createDropdown($baseTemplate, boardValue, boardPosition, options) {
 65         var $template = $baseTemplate.clone(true),
 66             $button = createButton(),
 67             $choices = createList();
 68         detachedChoices.push({button: $button, choices: $choices});
 69         return $template;
 70 
 71         /**
 72          * Creates a box to be placed at the board
 73          */
 74         function createButton() {
 75             var $button = $template.find("button");
 76             $button.attr('data-background', boardValue);
 77             $button.attr('data-position', boardPosition);
 78             return $button;
 79         }
 80 
 81         /**
 82          * Creates the list of available options for the player to be placed at the board after every button
 83          */
 84         function createList() {
 85             var $ul = $template.find(".dropdown-menu"),
 86                 $li = $ul.find(".dropdown-option").detach();
 87             for (var i = 0; i < options; i++) {
 88                 var $newLi = $li.clone(),
 89                     $a = $newLi.find(".choice");
 90                 $a.attr('data-background', i);
 91                 $a.attr('data-position', boardPosition);
 92                 $ul.append($newLi);
 93             }
 94             return $ul.detach();
 95         }
 96     }
 97 
 98 
 99     /**
100      * Listens for a player clicking on a choice and updates the value at the corresponding box
101      */
102     function performMove() {
103         $("body").on('click', '.choice', function(e) {
104             e.preventDefault();
105             var newValue = $(this).data('background'),
106                 position = $(this).data('position');
107             setPlayerConfig(position, newValue);
108             $gameBoardContainer.find(".game-button[data-position=" + position + "]").attr('data-background', newValue);
109             var $boxes = $gameBoardContainer.find(".game-button"),
110                 foundEmptyBox = false;
111             for (var i = 0; i < $boxes.length; i++) {
112                 if ($($boxes[i]).attr('data-background') == '-1') {
113                     foundEmptyBox = true;
114                     break;
115                 }
116             }
117             if (!foundEmptyBox) $(".submit-move").removeClass('disabled').removeAttr('disabled');
118         });
119     }
120 
121     /**
122      * Submits the player's board to the other players
123      * @param {object} sockets Instance of Sockets
124      */
125     function submitMove(sockets) {
126         $("body").on('click', '.submit-move', function(e) {
127             e.preventDefault();
128             var board = [];
129             $gameBoardContainer.find(".game-button").each(function() {
130                 board.push([$(this).attr('data-position'), parseInt($(this).attr('data-background'))]);
131             });
132             sockets.newMove(board);
133             $otherPlayersBoard.find(".box.transparent").removeClass('transparent'); //Show the other players boards
134         });
135     }
136 };
