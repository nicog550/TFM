1 "use strict";
  2 /**
  3  * Class responsible for the sockets
  4  * @constructor
  5  */
  6 var Sockets = function() {
  7     var ioSocket = io(),
  8         didAbandonGame = false,
  9         game,
 10         gameOver,
 11         messages = {
 12             ADD_USER: 'add user',
 13             FINAL_SCORES: 'final scores',
 14             GAME_OVER: 'game over',
 15             INVALID_LOGIN: 'invalid login',
 16             LOGIN: 'login',
 17             LOGOUT: 'logout',
 18             NEW_GAME: 'new game',
 19             NEW_MOVE: 'new move',
 20             REMAINING_PLAYERS_CHANGE: 'remaining players'
 21         },
 22         main,
 23         waitingRoom,
 24         welcome;
 25     return {
 26         init: function(mainRef, gameRef, gameOverRef, waitingRoomRef, welcomeRef) {
 27             main = mainRef;
 28             game = gameRef;
 29             gameOver = gameOverRef;
 30             waitingRoom = waitingRoomRef;
 31             welcome = welcomeRef;
 32             connect();
 33         },
 34         login: login,
 35         logout: logout,
 36         newMove: sendMove
 37     };
 38 
 39     /**
 40      * Connection to the room
 41      */
 42     function connect() {
 43         ioSocket.on(messages.LOGIN, function(data) {
 44             main.toggleScreen(waitingRoom);
 45             if (data.remainingPlayers == -1) waitingRoom.displayWaitForNextGameMessage();
 46             else waitingRoom.displayRemainingPlayers(data.remainingPlayers);
 47         });
 48 
 49         ioSocket.on(messages.NEW_GAME, function(data) {
 50             if (!didAbandonGame && (!game.debug() || $("#" + game.selector).hasClass('hidden')))
 51                 game.startGame(data.board, data.gameDuration, data.options, data.myName, data.otherPlayers,
 52                     data.showDuring, data.css);
 53         });
 54 
 55         ioSocket.on(messages.NEW_MOVE, function(data) {
 56             if (!didAbandonGame) game.drawMove(data.username, data.board);
 57         });
 58 
 59         ioSocket.on(messages.REMAINING_PLAYERS_CHANGE, function(data) {
 60             waitingRoom.displayRemainingPlayers(data.remainingPlayers);
 61         });
 62 
 63         ioSocket.on(messages.GAME_OVER, function(data) {
 64             if (!didAbandonGame && !game.debug() && !$("#" + game.selector).hasClass('hidden'))
 65                 game.finishGame(data.waitingTime);
 66         });
 67 
 68         ioSocket.on(messages.FINAL_SCORES, function(data) {
 69             if (!didAbandonGame) gameOver.displayFinalScores(data);
 70         });
 71 
 72         ioSocket.on(messages.INVALID_LOGIN, function(data) {
 73             welcome.invalidLogin(data.reason);
 74         });
 75     }
 76 
 77     /**
 78      * Performs the player's login
 79      */
 80     function login() {
 81         _send(messages.ADD_USER, main.getUsername());
 82     }
 83 
 84     /**
 85      * Performs the player's logout
 86      */
 87     function logout() {
 88         didAbandonGame = true;
 89         _send(messages.LOGOUT, main.username);
 90     }
 91 
 92     /**
 93      * Sends a new move
 94      * @param {Array} board The player's board
 95      */
 96     function sendMove(board) {
 97         _send(messages.NEW_MOVE, board);
 98     }
 99 
100     /**
101      * Sends a message through the socket
102      * @param {string} code The code of the message to emit
103      * @param {object} message The message to be emitted
104      * @private
105      */
106     function _send(code, message) {
107         ioSocket.emit(code, message);
108     }
109 };
