1 "use strict";
  2 /**
  3  * Class responsible for drawing the other players boards
  4  * @param {boolean} debugGame Game mode: debug or not
  5  * @constructor
  6  */
  7 var OtherPlayersBoard = function(debugGame) {
  8     var $otherPlayersBoard = $("#other-players-container");
  9     return {
 10         getBoard: function() { return $otherPlayersBoard; },
 11         displayBoards: displayBoards,
 12         drawMove: drawMove
 13     };
 14 
 15     /**
 16      * Empties the other players boards if they already had content and populates them again
 17      * @param {object} otherPlayers The other players boards
 18      */
 19     function displayBoards(otherPlayers) {
 20         if (debugGame && $otherPlayersBoard.children().length > 0) return;
 21         $otherPlayersBoard.empty();
 22         var $boardTemplate = $("#other-boards-template").find(".other-players-game").clone(),
 23             $playerTemplate = $boardTemplate.find(".other-players").detach();
 24         for (var player in otherPlayers) {
 25             if (otherPlayers.hasOwnProperty(player)) {
 26                 $boardTemplate.append(_createPlayerBoard($playerTemplate, player, otherPlayers[player]));
 27             }
 28         }
 29         $otherPlayersBoard.append($boardTemplate);
 30     }
 31 
 32     /**
 33      * Creates a board for one of the other players
 34      * @param {jQuery} $baseTemplate The template to be used for drawing the board
 35      * @param {string} playerName The name of the player
 36      * @param {Array} playerBoard The board of the player
 37      * @returns {jQuery}
 38      * @private
 39      */
 40     function _createPlayerBoard($baseTemplate, playerName, playerBoard) {
 41         var $player = $baseTemplate.clone(),
 42             $playerRow = $player.find(".player-data");
 43         $playerRow.attr('data-player', playerName);
 44         $player.find(".name").text(playerName);
 45         var $valueTemplate = $playerRow.find(".box").detach();
 46         fillPlayerData();
 47         return $player;
 48 
 49         function fillPlayerData() {
 50             for (var j = 0; j < playerBoard.length; j++) {
 51                 var $box = $valueTemplate.clone();
 52                 $box.attr('data-background', playerBoard[j]);
 53                 $box.attr('data-position', j);
 54                 $playerRow.append($box);
 55             }
 56 
 57         }
 58     }
 59 
 60     /**
 61      * Displays the move another player has done
 62      * @param {string} player The player's name
 63      * @param {Array} board The player's full board
 64      */
 65     function drawMove(player, board) {
 66         var $playerBoard = $("[data-player='" + player + "']");
 67         board.forEach(function(element) {
 68             $playerBoard.find("[data-position=" + element[0] + "]").attr('data-background', element[1]);
 69         });
 70     }
 71 };
