1 "use strict";
  2 /**
  3  * Class responsible for the welcome screen
  4  * @constructor
  5  */
  6 var Welcome = function() {
  7     var $button = $("#play-button"),
  8         $errorMessage = $("#invalid-login"),
  9         $input = $("#username"),
 10         main,
 11         game,
 12         sockets;
 13     return {
 14         init: function(mainRef, gameRef, socketsRef) {
 15             main = mainRef;
 16             game = gameRef;
 17             sockets = socketsRef;
 18             $button.on('click', _logPlayerIn);
 19             _loginOnEnter();
 20         },
 21         selector: "welcome-screen",
 22         invalidLogin: invalidLogin
 23     };
 24 
 25     /**
 26      * Displays the "already used username" or "room full" messages
 27      * @param {string} message The message to display
 28      */
 29     function invalidLogin(message) {
 30         $errorMessage.text(message);
 31         $errorMessage.removeClass('hidden');
 32         $button.removeAttr('disabled');
 33         $button.removeClass('disabled');
 34     }
 35 
 36     /**
 37      * Creates a random username
 38      * @returns {string}
 39      */
 40     function createUsername() {
 41         var text = "";
 42         var possible = "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789";
 43         for (var i = 0; i < 5; i++) text += possible.charAt(Math.floor(Math.random() * possible.length));
 44         return text;
 45     }
 46 
 47     /**
 48      * Checks if the player has inserted a name and, if so, notifies the socket to perform a login
 49      * @private
 50      */
 51     function _logPlayerIn() {
 52         var username = $input.val();
 53         if (username == "") {
 54             if (game.debug()) username = createUsername();
 55             else return;
 56         }
 57         $errorMessage.addClass('hidden');
 58         $button.attr('disabled', 'disabled');
 59         $button.addClass('disabled');
 60         main.setUsername(username);
 61         sockets.login();
 62     }
 63 
 64     /**
 65      * Triggers the _logPlayerIn function when the player presses enter at the username input
 66      * @private
 67      */
 68     function _loginOnEnter() {
 69         $input.on("keyup", function(event) {
 70             if (event.which == 13 && $(this).val() !== '') _logPlayerIn();
 71         });
 72     }
 73 };
