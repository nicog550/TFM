1 "use strict";
  2 /**
  3  * Class responsible for the waiting room screen
  4  * @constructor
  5  */
  6 var WaitingRoom = function() {
  7     var game,
  8         main;
  9     return {
 10         init: function(mainRef, gameRef) {
 11             main = mainRef;
 12             game = gameRef;
 13         },
 14         selector: "waiting-room-screen",
 15         displayRemainingPlayers: displayRemainingPlayers,
 16         displayWaitForNextGameMessage: displayWaitForNextGameMessage
 17     };
 18 
 19     /**
 20      * Displays the number of players still needed for the game to start
 21      * @param {number} remaining The remaining number of players
 22      */
 23     function displayRemainingPlayers(remaining) {
 24         $("#remaining-players-count").text(remaining);
 25     }
 26 
 27     /**
 28      * Notifies the players that they have entered the experiment after it has already started and that they will have
 29      * to wait until the next game starts to join it
 30      */
 31     function displayWaitForNextGameMessage() {
 32         $("#remaining-players").addClass('hidden');
 33         $("#wait-for-next-game").removeClass('hidden');
 34     }
 35 };
