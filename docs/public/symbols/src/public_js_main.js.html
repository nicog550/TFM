1 "use strict";
  2 /**
  3  * Main class, application launcher
  4  * @constructor
  5  */
  6 var Main = function() {
  7     var game = new Game(),
  8         gameOver = new GameOver(),
  9         sockets = new Sockets(),
 10         username,
 11         waitingRoom = new WaitingRoom(),
 12         welcomeScreen = new Welcome();
 13     /**
 14      * Initial tasks
 15      */
 16     return {
 17         init: function() {
 18             sockets.init(this, game, gameOver, waitingRoom, welcomeScreen);
 19             game.init(this, sockets, gameOver, waitingRoom);
 20             waitingRoom.init(this, game);
 21             welcomeScreen.init(this, game, sockets);
 22             toggleScreen(welcomeScreen, function() {
 23                 $("#username").trigger("focus");
 24             });
 25             _animateButtonsOnClick();
 26         },
 27         displayCountdown: displayCountdown,
 28         setUsername: setUsername,
 29         toggleScreen: toggleScreen,
 30         getUsername: getUsername
 31     };
 32 
 33     /**
 34      * Changes the buttons style when they are pressed in order to provide the user with visual feedback
 35      * @private
 36      */
 37     function _animateButtonsOnClick() {
 38         $("body").on('mousedown', ".btn", function() {
 39             $(this).addClass('clicked');
 40         }).on('mouseup', ".btn", function() {
 41             $(this).removeClass('clicked');
 42         });
 43     }
 44     
 45     function getUsername() {
 46         return username;
 47     }
 48 
 49     function setUsername(newName) {
 50         username = newName;
 51     }
 52 
 53     /**
 54      * Displays a countdown of seconds
 55      * @param {number} startTime Timeout starting seconds
 56      * @param {jQuery} $target jQuery object where the timeout will be displayed
 57      */
 58     function displayCountdown(startTime, $target) {
 59         updateDOM();
 60         var interval = setInterval(function() {
 61             updateDOM();
 62             if (startTime < 0) clearInterval(interval);
 63         }, 1000);
 64 
 65         function updateDOM() {
 66             $target.text(startTime--);
 67         }
 68     }
 69 
 70     /**
 71      * Hides the currently visible screen and replaces it with another one
 72      * @param {object} newOne Class from which to display the screen
 73      * @param {function} [callback] Callback to be called after switching screens
 74      */
 75     function toggleScreen(newOne, callback) {
 76         $(".screen:not(.hidden)").fadeOut(function() {
 77             $(this).addClass('hidden');
 78             $("#" + newOne.selector).removeClass('hidden').fadeIn();
 79             if (callback) callback();
 80         });
 81     }
 82 };
 83 
 84 $(document).ready(function() {
 85     Main().init();
 86 });
