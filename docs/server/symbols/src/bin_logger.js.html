1 "use strict";
  2 /**
  3  * Class responsible for writing lines at the CSV logs
  4  * @constructor
  5  */
  6 var Logger = function() {
  7     var fs = require('fs'),
  8         line,
  9         logFile;
 10     return {
 11         startGame: startGame,
 12         logAbandonment: logAbandonment,
 13         logMove: writeGameLine
 14     };
 15 
 16     /**
 17      * Logs the game settings and the game word plus each player's initial word
 18      * @param {number|string} round The current round
 19      * @param {string} word The game word
 20      * @param {Array} playersWords The initial words of all players
 21      * @param {string|number} players The number of players
 22      * @param {string|number} degree The small-world connection degree
 23      * @param {string|number} rewiring The rewiring probability
 24      * @param {string|number} letters The number of different letters
 25      * @param {string|number} wordLength The length of the word
 26      * @param {string|number} shownLetters The amount of letters shown to each player
 27      * @param {string|number} duration The duration of the game
 28      * @param {string|number} showCodeTime The amount of time the players see their codes before starting to play
 29      */
 30     function startGame(round, word, playersWords, players, degree, rewiring, letters, wordLength, shownLetters,
 31                        duration, showCodeTime) {
 32         logFile = _createFileName(round);
 33         _writeParameters(players, degree, rewiring, letters, wordLength, shownLetters, duration, showCodeTime);
 34         _writeColumnNames(true);
 35         writeGameLine(0, round, word);
 36         playersWords.forEach(function(current) { writeGameLine(current.userId, round, current.board); });
 37         _writeColumnNames(false);
 38         _logConnections(round, playersWords);
 39     }
 40 
 41     /**
 42      * Creates a filename for a new log
 43      * @param {string|number} round The current round
 44      * @returns {string}
 45      * @private
 46      */
 47     function _createFileName(round) {
 48         return 'gameLogs/' + _getDateTime() + ' - Round ' + round + '.csv';
 49     }
 50 
 51     /**
 52      * Returns the current datetime as yyyy-mm-dd hh:mm:ss
 53      * @returns {string}
 54      * @private
 55      */
 56     function _getDateTime() {
 57         var now = new Date();
 58         return now.getFullYear() + '-' + _zeroPad(now.getMonth() + 1) + '-' + _zeroPad(now.getDate()) + ' ' +
 59             _zeroPad(now.getHours()) + ':' + _zeroPad(now.getMinutes()) + ':' + _zeroPad(now.getSeconds());
 60 
 61         /**
 62          * Adds a leading zero to a number if it has only one digit
 63          * @param {number} integer The number to use
 64          * @returns {string}
 65          * @private
 66          */
 67         function _zeroPad(integer) {
 68             return (integer < 10 ? '0' : '') + integer.toString();
 69         }
 70     }
 71 
 72     /**
 73      * Logs the game settings
 74      * @param {string|number} players The number of players
 75      * @param {string|number} degree The small-world connection degree
 76      * @param {string|number} rewiring The rewiring probability
 77      * @param {string|number} letters The number of different letters
 78      * @param {string|number} wordLength The length of the word
 79      * @param {string|number} shownLetters The amount of letters shown to each player
 80      * @param {string|number} duration The duration of the game
 81      * @param {string|number} showCodeTime The amount of time the players see their codes before starting to play
 82      * @private
 83      */
 84     function _writeParameters(players, degree, rewiring, letters, wordLength, shownLetters, duration, showCodeTime) {
 85         fs.appendFileSync(logFile,
 86             ['Game setup: line#', 'SecondsInitialCodeShown', 'Players', 'deg', 'pr', '#ofLetters', 'WordLength',
 87                 '#LettersShown', 'SecondsOfGame'].join(',') + "\n" +
 88             [1, showCodeTime, players, degree, rewiring, letters, wordLength, shownLetters, duration].join(',') + "\n");
 89     }
 90 
 91     /**
 92      * Writes the column headers
 93      * @param {boolean} initial Whether the headers correspond to the initial configurations or to moves made by players
 94      * @private
 95      */
 96     function _writeColumnNames(initial) {
 97         fs.appendFileSync(logFile,
 98             ['Game ' + (initial ? 'initial code' : 'play') + ' line#',
 99                 'userID', 'round', 'actiontime', 'word'].join(',') + "\n"
100         );
101         line = 0;
102     }
103 
104     /**
105      * Logs a row
106      * @param {number|string} userId The player's ID
107      * @param {number|string} round The current round
108      * @param {Array} word The word to be logged
109      */
110     function writeGameLine(userId, round, word) {
111         fs.appendFileSync(logFile, ([++line, userId, round, _getDateTime()].concat(word)).join(',') + '\n');
112     }
113 
114     /**
115      * Logs the connections between players in a separate log file
116      * @param {string|number} round The current round
117      * @param {Array} playersWords The initial words of all players
118      * @private
119      */
120     function _logConnections(round, playersWords) {
121         var csvContent = 'userID,round,connections\n';
122         playersWords.forEach(function(current) {
123             csvContent += ([current.userId, round].concat(current.connections)).join(',') + '\n';
124         });
125         fs.appendFileSync(logFile.replace('.', ': connections.'), csvContent);
126     }
127 
128     /**
129      * Logs the moment in which a player abandons the game
130      * @param {string|number} round The current round
131      * @param {string|number} userId The player's ID
132      */
133     function logAbandonment(round, userId) {
134         if (logFile === undefined) {
135             logFile = _createFileName(0);
136         }
137         fs.appendFileSync(logFile.replace('.', ': abandonments.'),
138                           'userID,round,actiontime\n' + [userId, round, _getDateTime()].join(',') + '\n');
139     }
140 };
141 
142 module.exports = new Logger();
