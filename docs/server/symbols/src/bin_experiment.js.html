1 "use strict";
  2 
  3 /**
  4  * Class responsible for the creation of new games
  5  * @constructor
  6  */
  7 var Experiment = function() {
  8     var constants = require('./constants'),
  9         currentGame,
 10         experimentHasStarted = false,
 11         generator = require('./gameCore/generator'),
 12         logger = require('./logger'),
 13         playersBoards,
 14         round = 0,
 15         scores = require('./gameCore/scores'),
 16         sockets = [],
 17         word;
 18     return {
 19         init: function() {
 20             generator.init(getCurrentGame, getSockets, setWord, constants.players, logger);
 21             scores.init(constants.scores.pointsPerHit, constants.scores.bonusOnCompletedWord, getWord);
 22             currentGame = constants.games[round];
 23         },
 24         addPlayer: addPlayer,
 25         removePlayer: removePlayer,
 26         updatePlayerBoard: updatePlayerBoard
 27     };
 28 
 29     /** Getter */
 30     function getCurrentGame() { return currentGame; }
 31 
 32     /** Getter */
 33     function getSockets() { return sockets; }
 34 
 35     /** Getter */
 36     function getWord() { return word; }
 37 
 38     /** Setter */
 39     function setWord(newWord) { word = newWord; }
 40 
 41     /**
 42      * Function responsible for the generation of a new game
 43      */
 44     function _generateGame() {
 45         playersBoards = generator.createGame(round);
 46         setTimeout(_endGame, (currentGame.duration + currentGame.showCodeTime) * 1000);
 47     }
 48 
 49     /**
 50      * Sends the 'game over' message to all players
 51      * @private
 52      */
 53     function _endGame() {
 54         var allExperimentsDone = round + 1 == constants.games.length;
 55         sockets.forEach(function(socket) {
 56             socket.emit('game over', {waitingTime: (allExperimentsDone ? 0 : constants.intervalBetweenGames)});
 57         });
 58         var newScores = scores.calculateScores(word, playersBoards);
 59         sockets.forEach(function(socket) { socket.emit('final scores', newScores); });
 60         if (!allExperimentsDone) {
 61             currentGame = constants.games[++round];
 62             setTimeout(_generateGame, constants.intervalBetweenGames * 1000);
 63         }
 64     }
 65 
 66     /**
 67      * Appends the socket of a newly connected player to the list of existing sockets
 68      * @param socket The socket of the newly connected user
 69      * @returns {number|boolean} Three possibilities:<br>
 70      * 1. {number} Normal case: the number of remaining players necessary for the game to start<br>
 71      * 2. {boolean} True if the player has been added after the experiment has started<br>
 72      * 3. {boolean} False if no more players can be added or the experiment has ended<br>
 73      */
 74     function addPlayer(socket) {
 75         if (sockets.length == constants.players || round + 1 == constants.games.length) return false; //Case 3
 76 
 77         sockets.push(socket);
 78         scores.addPlayer(socket.username);
 79         var remainingNeededPlayers = constants.players - sockets.length;
 80 
 81         if (experimentHasStarted) { //Cases 2 or 3
 82             //The game has started but there is room for more players. The player will be able to join the room unless
 83             // the current round is the last one
 84             return round + 1 < constants.games.length;
 85         }
 86         //Case 1
 87         if (remainingNeededPlayers == 0) {
 88             experimentHasStarted = true;
 89             _generateGame();
 90         }
 91         return remainingNeededPlayers;
 92     }
 93 
 94     /**
 95      * Removes the socket of a disconnected player from the list of existing sockets
 96      * @param socket The socket of the disconnected user
 97      * @returns {boolean|number} The number of remaining players necessary for the game to start if the player was
 98      * already connected, false otherwise
 99      */
100     function removePlayer(socket) {
101         for (var i = 0; i < sockets.length; i++) {
102             if (sockets[i].userId == socket.userId) {
103                 sockets.splice(i, 1);
104                 logger.logAbandonment(round + 1, socket.userId);
105                 return constants.players - sockets.length;
106             }
107         }
108         return false;
109     }
110 
111     /**
112      * Updates a player's board
113      * @param {string} player The player's username
114      * @param {Array} board The player's board as [[position, value], [position, value], ...]
115      */
116     function updatePlayerBoard(player, board) {
117         for (var i = 0; i < playersBoards.length; i++) {
118             if (playersBoards[i].player == player) {
119                 board.forEach(function(positionAndValue, index) {
120                     playersBoards[i].board[index] = positionAndValue[1];
121                 });
122                 logger.logMove(playersBoards[i].userId, round + 1, board.map(function(positionAndValue) {
123                     return positionAndValue[1];
124                 }));
125                 break;
126             }
127         }
128     }
129 };
130 
131 module.exports = new Experiment();
