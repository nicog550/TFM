1 "use strict";
  2 /**
  3  * Class responsible for the creation of a new game
  4  * @constructor
  5  */
  6 var Generator = function() {
  7     var game,
  8         getCurrentGame,
  9         getSockets,
 10         graphGenerator = require('randomgraph'),
 11         logger,
 12         players,
 13         setWord;
 14     return {
 15         init: function(getCurrentGameRef, getSocketsRef, setWordRef, playersVal, loggerRef) {
 16             getCurrentGame = getCurrentGameRef;
 17             getSockets = getSocketsRef;
 18             setWord = setWordRef;
 19             players = playersVal;
 20             logger = loggerRef;
 21         },
 22         createGame: createGame
 23     };
 24 
 25     /**
 26      * Algorithm for the creation of a new game
 27      * @param {string|number} round The current round
 28      * @returns {Array} Each player's game. The structure looks like
 29      * [{player: str, userId: str, board: []}, {player: str, userId: str, board: []}, ...]
 30      */
 31     function createGame(round) {
 32         game = getCurrentGame();
 33         var wordLength = game.wordLength,
 34             gameWord = _generateWord(wordLength),
 35             shownLetters = Math.ceil(wordLength / players), //Number of correct letters that each player will see
 36             generatedGames = [],
 37             connections = _generateConnectionsBetweenPlayers(players, game.degree, game.rewiring);
 38         for (var i = 0; i < players; i++) {
 39             generatedGames.push(_generateGameForPlayer(gameWord.slice(), shownLetters));
 40         }
 41         var css = _generateGameColors(game.options),
 42             playersWords = _sendGamesToPlayers(generatedGames, connections, css);
 43         logger.startGame(round + 1, gameWord, playersWords, players, game.degree, game.rewiring, game.options,
 44                          wordLength, shownLetters, game.duration, game.showCodeTime);
 45         return playersWords;
 46     }
 47 
 48     /**
 49      * Generates a random code, which will be the word used at the new game
 50      * @param {number} wordLength The length of the word to be generated
 51      * @returns {Array} An array of <i>constants.wordLength</i> positions
 52      */
 53     function _generateWord(wordLength) {
 54         var word = [];
 55         for (var i = 0; i < wordLength; i++) word.push(Math.floor(Math.random() * game.options));
 56         setWord(word);
 57         return word;
 58     }
 59 
 60     /**
 61      * Generates the graph which represents the players that each player will see. Each position in the array represents
 62      * a player, so, in the following example, the third player is the one that will see 0, 1 and 4.
 63      * [[1, 5, 6], [2, 7, 8], [0, 1, 4], ...]
 64      * @param {number} numberOfPlayers The number of players for the new game
 65      * @param {number} degree The number of players that each player will see
 66      * @param {number} rewiring The rewiring probability
 67      * @returns {Array}
 68      * @private
 69      */
 70     function _generateConnectionsBetweenPlayers(numberOfPlayers, degree, rewiring) {
 71         var connections = graphGenerator.WattsStrogatz.beta(numberOfPlayers, degree, rewiring),
 72             formattedConnections = [];
 73         connections.edges.forEach(function(edge) {
 74             if (formattedConnections[edge.source] === undefined) formattedConnections[edge.source] = [];
 75             formattedConnections[edge.source].push(edge.target);
 76         });
 77         return formattedConnections;
 78     }
 79 
 80     /**
 81      * Generates the word that will contain the real letters the player will see
 82      * @param {Array} gameWord The game's word
 83      * @param {number} lettersToShow The number of positions that the player will see
 84      * @returns {*}
 85      * @private
 86      */
 87     function _generateGameForPlayer(gameWord, lettersToShow) {
 88         var positionsThatPlayerWillSee = setPositionsToShow();
 89         gameWord.forEach(function(letter, position) {
 90             if (positionsThatPlayerWillSee.indexOf(position) === -1) gameWord[position] = -1;
 91         });
 92         return gameWord;
 93 
 94         /**
 95          * Chooses the random positions of the game word that the player will see
 96          * @returns {Array}
 97          */
 98         function setPositionsToShow() {
 99             var positions = [],
100                 choice;
101             while (positions.length < lettersToShow) {
102                 choice = Math.floor(Math.random() * (gameWord.length - 1));
103                 //If the choice wasn't already in 'positions', append it
104                 if (positions.indexOf(choice) === -1) positions.push(choice);
105             }
106             return positions;
107         }
108     }
109 
110     /**
111      * Generates the CSS sheet with the colors that players will see for each choice
112      * @param {number} options The number of different choices
113      * @private
114      */
115     function _generateGameColors(options) {
116         var colors = [],
117             values = ['0', '5', 'B', 'F'];
118         values.forEach(function(val1) {
119             values.forEach(function(val2) {
120                 values.forEach(function(val3) {
121                     colors.push('#' + val1 + val2 + val3);
122                 });
123             });
124         });
125         colors.shift(); //Remove the first element (#000), which would be confused with the background color
126         var finalValues = [];
127         for (var i = 0; i < options; i++) {
128             finalValues.push(colors.splice(Math.floor(Math.random() * colors.length), 1));
129         }
130         return finalValues.reduce(function(accum, current, index) {
131             //E.g.: [data-background="1"], [data-background="1"]:hover { background-color: #5FB !important; }
132             return accum +
133                 '[data-background="' + index + '"], [data-background="' + index + '"]:hover {' +
134                 ' background-color: ' + current + ' !important; ' +
135                 '}\n';
136         }, '');
137     }
138 
139     /**
140      * Sends to each player their game plus the games of the other players. The used structure is the following one:
141      * <pre><code>
142      * {
143      *      gameDuration: number,
144      *      options: number,
145      *      board: [...],
146      *      myName: string,
147      *      otherPlayers: {username1: [...], username2: [...], ...},
148      *      showDuring: number,
149      *      css: string
150      * }
151      * </code></pre>
152      * @private
153      * @param {Array} generatedGames The games generated for all players
154      * @param {Array} connections The connections generated for all players
155      * @return {Array} The games generated for each player, indexed by their name
156      * @param {string} css The CSS sheet with the colors for each option
157      */
158     function _sendGamesToPlayers(generatedGames, connections, css) {
159         var sockets = getSockets(),
160             initialGames = [];
161         sockets.forEach(function(socket, index) {
162             socket.emit('new game', {
163                 gameDuration: game.duration,
164                 options: game.options,
165                 board: generatedGames[index],
166                 myName: socket.username,
167                 otherPlayers: getOtherPlayersBoards(connections[index]),
168                 showDuring: game.showCodeTime,
169                 css: css
170             });
171             initialGames.push({
172                 player: socket.username,
173                 userId: socket.userId,
174                 board: generatedGames[index],
175                 connections: connections[index]
176             });
177         });
178         return initialGames;
179 
180         function getOtherPlayersBoards(playerConnections) {
181             var board = {};
182             playerConnections.forEach(function(player) {
183                 board[sockets[player].username] = generatedGames[player];
184             });
185             return board;
186         }
187     }
188 };
189 
190 module.exports = new Generator();
