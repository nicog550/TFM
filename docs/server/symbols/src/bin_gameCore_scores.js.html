1 "use strict";
  2 /**
  3  * Class responsible for the management of the game scores
  4  * @constructor
  5  */
  6 var Scores = function() {
  7     var bonusOnCompletedWord,
  8         getWord,
  9         pointsPerHit,
 10         scores = {};
 11     return {
 12         init: function(pointsPerHitVal, bonusOnCompletedWordVal, getWordRef) {
 13             pointsPerHit = pointsPerHitVal;
 14             bonusOnCompletedWord = bonusOnCompletedWordVal;
 15             getWord = getWordRef;
 16         },
 17         addPlayer: addPlayer,
 18         calculateScores: calculateScores
 19     };
 20 
 21     /**
 22      * Adds a new player to the scores
 23      * @param {string} playerName The player's name
 24      */
 25     function addPlayer(playerName) {
 26         scores[playerName] = 0;
 27     }
 28 
 29     /**
 30      * Calculates the scores of all players and returns them sorted
 31      * @param {Array} word The current game's word
 32      * @param {Array} playersBoards The boards of all players in the following form:
 33      * [{player: str, board: []}, {player: str, board: []}, ...]
 34      * @returns {Array}
 35      */
 36     function calculateScores(word, playersBoards) {
 37         var missed,
 38             playerPoints,
 39             roundScores = [];
 40         playersBoards.forEach(function(current) {
 41             missed = false;
 42             playerPoints = 0;
 43             current.board.forEach(function(valueInBoard, index) {
 44                 if (word[index] == valueInBoard) playerPoints += pointsPerHit;
 45                 else missed = true;
 46             });
 47             if (!missed) playerPoints += bonusOnCompletedWord;
 48             scores[current.player] += playerPoints;
 49             roundScores.push({username: current.player, round: playerPoints, total: scores[current.player]});
 50         });
 51         return _sortScores(_includeScoresOfPlayersThatAbandonedTheGame(roundScores));
 52     }
 53 
 54     /**
 55      * Checks if exist scores of players which have abandoned the game and appends them to the final scores of the
 56      * current round
 57      * @param {Array} currentScores The final scores of the current round
 58      * @returns {Array}
 59      * @private
 60      */
 61     function _includeScoresOfPlayersThatAbandonedTheGame(currentScores) {
 62         var i, found, scoresOfOldPlayers = [];
 63         for (var player in scores) {
 64             if (scores.hasOwnProperty(player)) {
 65                 found = false;
 66                 for (i = 0; i < currentScores.length; i++) {
 67                     if (currentScores[i].username == player) {
 68                         found = true;
 69                         break;
 70                     }
 71                 }
 72                 if (!found) scoresOfOldPlayers.push({username: player, round: 0, total: scores[player]});
 73             }
 74         }
 75         return currentScores.concat(scoresOfOldPlayers);
 76     }
 77 
 78     /**
 79      * Sorts the scores from the one with the highest total to the one with the lowest total
 80      * @param {Array} roundScores The scores at the current round
 81      * @returns {Array}
 82      * @private
 83      */
 84     function _sortScores(roundScores) {
 85         roundScores.sort(function(a, b) {return b.total - a.total});
 86         return roundScores;
 87     }
 88 
 89 };
 90 
 91 module.exports = new Scores();
